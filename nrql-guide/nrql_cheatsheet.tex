\sect{TBox}
\begin{lstlisting}[language=nrql]
(implies     Dog Animal)        ; Dog<=Animal
(equivalent  Mother
   (and Woman (some has-child Human)))
(disjoint    Man Woman)
(define-primitive-role has-child
   :inverse has-parent :transitive nil
   :domain Human :range Human)
(inverse     withdrawal deposit)
(implies-role deposit transaction)
(define-concrete-domain-attribute
   age :type integer)
\end{lstlisting}

\sect{ABox}
\begin{lstlisting}[language=nrql]
(instance    alice Woman)
(related     alice junior has-child)
(constrained alice a-age age)
(constraints (equal a-age 50)) ; int->equal
(different-from k1 k2)         ; no UNA!
\end{lstlisting}
{\scriptsize CD preds: int \code{(equal a v)}, real \code{(= a v)(> a v)}, string \code{(string= a s)}.}

\sect{Variables}
\begin{lstlisting}[language=nrql]
?x        ; ABox var, NON-injective
$?x       ; ABox var, INJECTIVE  (distinct!)
?*x  $?*x ; substrate node (non-inj / inj)
*ind      ; a named substrate individual
\end{lstlisting}
{\scriptsize \textbf{NB:} reverse of the UG (RacerPro 2.0 was changed; \code{\$?} is injective). Force distinctness explicitly --- see Gotchas.}

\sect{Killer idioms (live-verified)}
\begin{lstlisting}[language=nrql]
;; sh:minCount 1  (= NOT EXISTS)
(retrieve (?x) (and (?x Person)
  (neg (project-to (?x) (?x ?y fn)))))
;; sh:class on a path  (no counterexample)
(retrieve (?x) (and (?x Parent)
  (?x ?y has-child) (neg (?y Person))))
;; sh:maxCount 1  (no UNA: over-reports)
(retrieve (?x) (and (?x Person)
  (?x $?a fn) (?x $?b fn)
  (neg (same-as $?a $?b))))
;; sh:disjoint  (shared value)
(retrieve (?x)
  (and (?x ?v likes) (?x ?v dislikes)))
\end{lstlisting}

\sect{MiniLisp hatch (termination-safe!)}
\begin{lstlisting}[language=nrql]
(evaluate (+ 1 2))    ; restricted evaluator
;; retrieve1 head with a
;;   (:lambda (x) .. :reject) = a filter
\end{lstlisting}
{\scriptsize \textbf{Has} arith, \code{format}, \code{search}, \code{subseq}, \code{reduce}, \code{dotimes}, KB callbacks. \textbf{Lacks} recursion (aborted to ensure termination), \code{loop}, \code{require}, regex. Not Turing-complete --- by design, for a decidable reasoner.}

\sect{\color{acc}Gotchas}
{\scriptsize
$\bullet$ \textbf{No UNA}: \code{(at-most 1 R)} + 2 named fillers is \emph{consistent} until \code{(different-from..)}.\\
$\bullet$ \code{(neg (?x ?y R))} $\ne$ \code{(neg (project-to (?x)(?x ?y R)))} --- use the 2nd for "no R-successor".\\
$\bullet$ \code{neg} = failure over \emph{known} objects, not classical negation, not "triple absent".\\
$\bullet$ \code{has-known-successor} sees \emph{named} successors; an OWL existential's anonymous witness it won't.\\
$\bullet$ \textbf{Var inequality}: \code{(neg (same-as ?x ?y))} or \code{(neg (= ?x ?y))}. No \code{<>} for individuals; \code{different-from} is ABox-only. UNA toggling doesn't change query var-distinctness.}
